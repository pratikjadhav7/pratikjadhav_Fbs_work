Q3) Create jobs Table with Constraints
Write an SQL query to create a table named jobs including columns job_id, job_title,
min_salary, max_salary.
Apply constraints:
- job_id must be Primary Key
- job_title cannot be NULL
- max_salary must not exceed 25000

Query:mysql> create table jobs(
    -> job_id int primary key,
    -> job_title varchar(20) not null,
    -> min_salary decimal(4,2),
    -> max_salary decimal(7,2) check(max_salary<2500));

Q7) Create departments Table (Composite Primary Key)
Create Department table with following structure:
Field Name Type Null Key
department_id DECIMAL(4,0) NO PRI
department_name VARCHAR(30) NO
manager_id DECIMAL(6,0) NO PRI
location_id DECIMAL(4,0) YES
Composite Primary Key: (department_id, manager_id)

Query:mysql> create table departments(
    -> dept_id decimal(4,0) not null,
    -> dept_name varchar(30) not null,
    -> manager_id decimal(6,0) not null,
    -> location_id decimal(4,0),
    -> primary key(dept_id,manager_id));



Q8) Create employees Table with Normal Foreign Key
Write an SQL query to create a table named employees including columns:
employee_id, first_name, last_name, email, phone_number, hire_date, job_id,
salary, commission, manager_id and department_id.
Apply constraints:
- employee_id must be Primary Key
- department_id must be a Foreign Key referencing departments(department_id)

Query:mysql> create table employees(
    -> employee_id int primary key,
    -> first_name varchar(50),
    -> last_name varchar(50),
    -> email varchar(100),
    -> phone_no varchar(20),
    -> hire_date date,
    -> job_id int,
    -> salary decimal(10,2),
    -> commision decimal(10,2),
    -> manager_id int,
    -> dept_id decimal(4,0),
    -> foreign key(dept_id) references departments(dept_id));
