Q35) Write an SQL query to list members who have issued the same book as issued by 'Garima

FROM Member m

    SELECT i2.book_no
    FROM Issue i2
    JOIN Member m2 ON i2.member_id = m2.member_id
    WHERE m2.member_name = 'Garima Sen');

Q38) Write an SQL query to display member(s) who issued the maximum and minimum number
of books.

SELECT m.member_id, m.member_name, COUNT(*) total
FROM Member m
JOIN Issue i ON m.member_id = i.member_id
GROUP BY m.member_id, m.member_name
HAVING total = (SELECT MAX(cnt) FROM (SELECT COUNT(*) cnt FROM Issue GROUP BY member_id) x)
    OR total = (SELECT MIN(cnt) FROM (SELECT COUNT(*) cnt FROM Issue GROUP BY member_id) x);

FROM Book b

GROUP BY b.book_no, b.book_name
HAVING COUNT(i.book_no) = (
    SELECT MAX(cnt)
    FROM (
        SELECT COUNT(book_no) AS cnt
        FROM Issue
        GROUP BY book_no)t);

Q49) Write an SQL query to list authors whose books are least issued by Lifetime members.

SELECT b.author, COUNT(*) AS issue_count
FROM Book b
JOIN Issue i ON b.book_no = i.book_no
JOIN Member m ON i.member_id = m.member_id
WHERE m.membership_type = 'Lifetime'
GROUP BY b.author
HAVING COUNT(*) = (
    SELECT MIN(cnt)
    FROM (
        SELECT COUNT(*) AS cnt
        FROM Book b2
        JOIN Issue i2 ON b2.book_no = i2.book_no
        JOIN Member m2 ON i2.member_id = m2.member_id
        WHERE m2.membership_type = 'Lifetime'
        GROUP BY b2.author) t);
